# Farmacia

Projeto para estudo de HTML, CSS, JS e PHP, utilizando a arquitetura MVC. 

### SQL do Projeto

```MySQL
CREATE SCHEMA FARMACIA;

CREATE TABLE FARMACIA.TBCARGO (
CARCODIGO SERIAL NOT NULL PRIMARY KEY,
CARNOME VARCHAR(50) NOT NULL
);

CREATE TABLE FARMACIA.TBCIDADE (
CIDCODIGO SERIAL NOT NULL PRIMARY KEY,
CIDNOME VARCHAR(50) NOT NULL
);

CREATE TABLE FARMACIA.TBFORNECEDOR (
FORCODIGO SERIAL NOT NULL PRIMARY KEY,
FORNOME VARCHAR(50) NOT NULL,
CIDCODIGO INT NOT NULL,
FOREIGN KEY (CIDCODIGO) REFERENCES FARMACIA.TBCIDADE(CIDCODIGO)
);

CREATE TABLE FARMACIA.TBMEDICAMENTO (
MEDCODIGO SERIAL NOT NULL,
MEDVALOR NUMERIC(6, 2) NOT NULL,
FORCODIGO INT NOT NULL,
PRIMARY KEY(MEDCODIGO),
FOREIGN KEY (FORCODIGO) REFERENCES FARMACIA.TBFORNECEDOR(FORCODIGO)
);
```
